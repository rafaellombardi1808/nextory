             onerror="this.replaceWith(Object.assign(document.createElement('div'), {className:'placeholder', innerHTML:'SEM FOTO AINDA<br><span style=\\'font-weight:900\\'>coloque em /img</span>'}));">`
        : `<div class="placeholder">SEM FOTO AINDA<br><span style="font-weight:900">coloque em /img</span></div>`;

      return `${badge}${img}`;
    }

    function productCardHTML(p){
      const out = isOutOfStock(p);

      const btnPrimary = out
        ? `<a class="btn btn-primary btn-mini" href="${buildNotifyLink(p.id)}" target="_blank" rel="noopener">🔔 Avise-me quando chegar</a>`
        : `<button class="btn btn-primary btn-mini" onclick="addToCart('${escapeTxt(p.id)}')">🛍️ </button>`;

      const btnSecond = `<button class="btn btn-mini btn-muted" onclick="openCart()">🛒 Ver carrinho</button>`;

      return `
        <article class="prod">
          <div class="prod-img">
            ${renderProductMedia(p)}
          </div>

          <div class="prod-body">
            <h3 class="prod-name">${escapeTxt(p.name)}</h3>
            <p class="prod-desc">${escapeTxt(p.desc || "")}</p>
            <div class="row">
              <span class="chip">${escapeTxt(p.category)}</span>
              <span class="price">${fmtBRL(p.price)}</span>
            </div>
          </div>

          <div class="actions">
            ${btnPrimary}
            ${btnSecond}
          </div>
        </article>
      `;
    }

    function renderCatalog(){
      const list = getFilteredProducts();
      gridEl.innerHTML = list.map(productCardHTML).join("");

      if(list.length === 0){
        gridEl.innerHTML = `
          <div class="card" style="grid-column:1/-1; padding:16px;">
            <b>Nenhum produto encontrado.</b>
            <div class="small" style="margin-top:6px; color:var(--muted); font-weight:800;">
              Tente outra busca na barra do topo ou mude a categoria no menu ☰.
            </div>
          </div>
        `;
      }
    }

    // ========= LINHAS / VITRINES =========
    function takeFirst(list, n){ return list.slice(0, n); }

    function renderLines(){
      // Linhas também respeitam busca + categoria ativa (fica coerente)
      const filtered = getFilteredProducts();

      const bisquit = takeFirst(filtered.filter(p=>p.category === "Bisquit"), 8);
      const escolar = takeFirst(filtered.filter(p=>p.category === "Escolar"), 8);
      const livrosJogos = takeFirst(filtered.filter(p=>p.category === "Livros" || p.category === "Jogos"), 8);
      const encomenda = takeFirst(filtered.filter(p=>p.category === "Sob encomenda"), 8);

      gridLineBisquit.innerHTML = bisquit.map(productCardHTML).join("") || `<div class="card" style="grid-column:1/-1;padding:14px;">Sem itens aqui ainda.</div>`;
      gridLineEscolar.innerHTML = escolar.map(productCardHTML).join("") || `<div class="card" style="grid-column:1/-1;padding:14px;">Sem itens aqui ainda.</div>`;
      gridLineLivrosJogos.innerHTML = livrosJogos.map(productCardHTML).join("") || `<div class="card" style="grid-column:1/-1;padding:14px;">Sem itens aqui ainda.</div>`;
      gridLineEncomenda.innerHTML = encomenda.map(productCardHTML).join("") || `<div class="card" style="grid-column:1/-1;padding:14px;">Sem itens aqui ainda.</div>`;
    }

    function renderAll(){
      renderLines();
      renderCatalog();
      updateCartUI();
      syncTopbarHeight();
    }

    // ========= CARROSSEL =========
    (function(){
      const track = document.getElementById("plTrack");
      const dotsWrap = document.getElementById("plDots");
      const dots = Array.from(dotsWrap.querySelectorAll(".pl-dot"));
      const TOTAL = 4;
      const DELAY = 9000;
      let index = 0;
      let timer = null;

      function setActiveDot(i){
        dots.forEach((d,k)=> d.classList.toggle("is-active", k===i));
      }
      function goTo(i){
        index = (i + TOTAL) % TOTAL;
        track.style.transform = `translate3d(-${index * 100}%, 0, 0)`;
        setActiveDot(index);
      }
      function next(){ goTo(index + 1); }
      function start(){
        stop();
        timer = setInterval(next, DELAY);
      }
      function stop(){
        if(timer){ clearInterval(timer); timer = null; }
      }

      dots.forEach((dot)=>{
        dot.addEventListener("click", ()=>{
          goTo(Number(dot.dataset.i));
          start();
        });
      });

      goTo(0);
      start();
    })();

    // ========= INIT =========
    setTheme(localStorage.getItem(THEME_KEY) || "light");
    setActiveChip();
    renderAll();
  </script>
