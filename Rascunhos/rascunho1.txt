/* ================= CAROUSEL ================= */

let currentSlide = 0;
const track = document.getElementById("carouselTrack");
const slides = document.querySelectorAll(".slide");
const dotsContainer = document.getElementById("carouselDots");

function updateCarousel(){
  track.style.transform = `translateX(-${currentSlide * 100}%)`;

  document.querySelectorAll(".carousel-dots span").forEach((dot,i)=>{
    dot.classList.toggle("active", i===currentSlide);
  });
}

function nextSlide(){
  currentSlide = (currentSlide + 1) % slides.length;
  updateCarousel();
}

function prevSlide(){
  currentSlide = (currentSlide - 1 + slides.length) % slides.length;
  updateCarousel();
}

slides.forEach((_,i)=>{
  const dot = document.createElement("span");
  dot.onclick = ()=>{
    currentSlide = i;
    updateCarousel();
  };
  dotsContainer.appendChild(dot);
});

updateCarousel();

// autoplay
setInterval(nextSlide, 5000);

        const priceTxt = p.price == null ? "Consulte" : fmtBRL(p.price);

      }

      const totalTxt = fmtBRL(cartTotal());
      const msg =
`Olá! Quero fazer um pedido na Pingo Lingo Shop 💙

Itens:
${items.join("\n")}

Total (itens com preço): ${totalTxt}

Observações:
- Se tiver item "Consulte", podem me informar o valor?
- Meu nome: ______
- Endereço/retirada: ______`;

      return `https://wa.me/${WHATSAPP_NUMBER}?text=${encodeURIComponent(msg)}`;
    }

    // ========= CATALOG / RENDER =========
    function getFilteredProducts(){
      const q = (searchInput.value || "").trim().toLowerCase();
      const cat = catSelect.value;

      let list = PRODUCTS.filter(p=>{
        const matchQ =
          !q ||
          p.name.toLowerCase().includes(q) ||
          (p.desc || "").toLowerCase().includes(q) ||
          (p.category || "").toLowerCase().includes(q);

        const matchCat = (cat === "all") || (p.category === cat);
        return matchQ && matchCat;
      });

      const sort = sortSelect.value;
      if(sort === "price-asc"){
        list.sort((a,b)=> (a.price ?? 1e9) - (b.price ?? 1e9));
      }else if(sort === "price-desc"){
        list.sort((a,b)=> (b.price ?? -1) - (a.price ?? -1));
      }else if(sort === "name-asc"){
        list.sort((a,b)=> a.name.localeCompare(b.name, "pt-BR"));
      }
      return list;
    }

    function renderProductMedia(p){
      const badge = isOutOfStock(p)
        ? `<div class="badge-top">⏳ Esgotado</div>`
        : (p.category === "Bisquit" ? `<div class="badge-top">🎨 Bisquit</div>` : `<div class="badge-top">${escapeTxt(p.category)}</div>`);

      const img = p.image
        ? `<img src="${escapeTxt(p.image)}" alt="${escapeTxt(p.name)}" loading="lazy"
             onerror="this.replaceWith(Object.assign(document.createElement('div'), {className:'placeholder', innerHTML:'SEM FOTO AINDA<br><span style=\\'font-weight:900\\'>coloque em /img</span>'}));">`
        : `<div class="placeholder">SEM FOTO AINDA<br><span style="font-weight:900">coloque em /img</span></div>`;

      return `${badge}${img}`;
    }

    (function(){
      const track = document.getElementById("plTrack");
      const dotsWrap = document.getElementById("plDots");
      const dots = Array.from(dotsWrap.querySelectorAll(".pl-dot"));
      const TOTAL = 4;
      const DELAY = 9000;
      let index = 0;
      let timer = null;

      function setActiveDot(i){
        dots.forEach((d,k)=> d.classList.toggle("is-active", k===i));
      }
      function goTo(i){
        index = (i + TOTAL) % TOTAL;
        track.style.transform = `translate3d(-${index * 100}%, 0, 0)`;
        setActiveDot(index);
      }
      function next(){ goTo(index + 1); }
      function start(){
        stop();
        timer = setInterval(next, DELAY);
      }
      function stop(){
        if(timer){ clearInterval(timer); timer = null; }
      }

      dots.forEach((dot)=>{
        dot.addEventListener("click", ()=>{
          goTo(Number(dot.dataset.i));
          start();
        });
      });

      goTo(0);
      start();
    })();
